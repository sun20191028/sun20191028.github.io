sql 功能
mysql -h 127.0.0.1 -u springboot -p 用户名、密码登录 ，按enter之后 ，再输入密码
mysql -P3308 -u root 登录某个端口
show master status; 查询主库的执行脚本文件名称和位置
show variables like ‘server_id’; 查看server_id
show global variables like ‘port’; 查看端口
select user,host from mysql.user; 查询用户
explain select * from xxl_job_group; 解释sql
show processlist; 查询连接 进程
select @@tx_isolation; 查询当前会话事务隔离级别 mysql 默认都是 REPEATABLE-READ
select @@global.tx_isolation; 查询mysql全局事务隔离级别
show session variables like ‘autocommit’; 查看当前会话自动提交状态
show global variables like ‘autocommit’; 查看全局自动提交状态
set session autocommit = 0; 当前会话 设置 自动提交状态 为 OFF ，1 的时候为ON
show global variables like ‘wait_timeout’; 查看连接超时时间
show global variables like ‘%timeout%’;
show variables like ‘%timeout%’;
